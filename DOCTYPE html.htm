const fmtR = v => v?Number(v).toLocaleString('ru-RU',{maximumFractionDigits:0}):'—';

const fmtP = v => (v!=null&&!isNaN(v))?v.toFixed(1)+'%':'—';

const toN  = v => (v===''||v==null)?null:Number(v);

const esc  = s => String(s??'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');

 

function calcT(t){

  const nmc=toN(t.nmc),price=toN(t.price),cost=toN(t.cost);

  const margin    = (price!=null&&cost!=null)?price-cost:null;

  const marginPct = (margin!=null&&price>0)?margin/price*100:null;

  const discount  = (price!=null&&nmc!=null&&nmc>0)?(1-price/nmc)*100:null;

  return{margin,marginPct,discount};

}

 

function mColor(p){ return p==null?'#6b7280':p<5?'#dc2626':p<15?'#d97706':'#15803d'; }

 

function chip(val,map){

  return `<span class="badge" style="background:${bg};color:${fg}">${esc(val)}</span>`;

}

// ── CRUD ──────────────────────────────────────────

async function load(){

  const{data,error}=await sb.from('tenders').select('*').order('created_at',{ascending:false});

  if(!error&&data)tenders=data;

  render();

}

 

async function insert(row){

  const{data,error}=await sb.from('tenders').insert([row]).select();

  if(!error&&data)tenders=[data[0],...tenders];

  render();

}

 

async function update(id,row){

  const{error}=await sb.from('tenders').update(row).eq('id',id);

  if(!error)tenders=tenders.map(x=>x.id===id?{...x,...row}:x);

  render();

}

 

async function remove(id){

  const{error}=await sb.from('tenders').delete().eq('id',id);

  if(!error)tenders=tenders.filter(x=>x.id!==id);

  render();

}

 

// ── Filtering & Stats ─────────────────────────────

function filtered(){

  return tenders.filter(t=>{

    if(filter.type     &&t.type!==filter.type)        return false;

    if(filter.status   &&t.status!==filter.status)    return false;

    if(filter.result   &&t.result!==filter.result)    return false;

    if(filter.employee &&t.employee!==filter.employee)return false;

    if(filter.q){

      const q=filter.q.toLowerCase();

      if(![t.number,t.name,t.customer,t.employee].some(s=>s&&s.toLowerCase().includes(q)))return false;

    }

    return true;

  });

}

 

function stats(){

  const byS=Object.fromEntries(STATUSES.map(s=>[s,0]));

  let nmcSum=0,mSum=0,mCnt=0,wins=0;

  tenders.forEach(t=>{

    if(t.status in byS)byS[t.status]++;

    if(t.result==='Победа')wins++;

    if(t.nmc)nmcSum+=Number(t.nmc);

    const{marginPct}=calcT(t);

    if(marginPct!=null){mSum+=marginPct;mCnt++;}

  });

  return{byS,wins,nmcSum,total:tenders.length,avgM:mCnt?mSum/mCnt:null};

}

      const mc=mColor(marginPct);

      const numEl=t.url

        ?`<a href="${esc(t.url)}" target="_blank" style="color:#2563eb">${esc(t.number||'—')}</a>`

        :esc(t.number||'—');

      const decCls=t.decision==='Участвуем'?'dec-yes':t.decision==='Не участвуем'?'dec-no':'';

      return`<tr>

        <td class="tbl-mono">${numEl}</td>

        <td class="tbl-name-col">

          <div class="tbl-name-main" title="${esc(t.name)}">${esc(t.name)||'—'}</div>

          <div class="tbl-name-sub">${esc([t.customer,t.platform].filter(Boolean).join(' · '))}</div>

        </td>

        <td>${chip(t.type,TYPE_C)}</td>

        <td class="tbl-right">${fmtR(t.nmc)}</td>

        <td class="tbl-right">${fmtR(t.price)}</td>

        <td class="tbl-center">${fmtP(discount)}</td>

        <td class="tbl-center fw6" style="color:${mc}">${fmtP(marginPct)}</td>

        <td>${chip(t.status,STATUS_C)}</td>

        <td class="${decCls}">${esc(t.decision||'—')}</td>

        <td>${chip(t.result,RESULT_C)}</td>

        <td style="color:#6b7280">${esc(t.employee||'—')}</td>

        <td style="color:#9ca3af;white-space:nowrap">${esc(t.deadline||'—')}</td>

        <td style="white-space:nowrap">

          <button class="icon-btn" style="color:#3b82f6" onclick="openEdit(${t.id})">✎</button>

          <button class="icon-btn" style="color:#f87171" onclick="openDelete(${t.id})">✕</button>

        </td>

      </tr>`;

    }).join('');

}

 

function sel(bindKey,opts,placeholder,attrs){

  const cur=eval(bindKey);

  return`<select ${attrs} onchange="${bindKey}=this.value;render()">

    <option value="">${placeholder}</option>

    ${opts.map(o=>`<option${cur===o?' selected':''}>${esc(o)}</option>`).join('')}

  </select>`;

}

 

function resetFilter(){filter={q:'',type:'',status:'',result:'',employee:''};render();}

 

// ── Modal ─────────────────────────────────────────

function formVals(){

  const g=id=>document.getElementById(id)?.value??'';

  return{number:g('f-num'),name:g('f-name'),customer:g('f-cust'),platform:g('f-plat'),

    url:g('f-url'),type:g('f-type'),nmc:g('f-nmc'),price:g('f-price'),cost:g('f-cost'),

    status:g('f-stat'),decision:g('f-dec'),result:g('f-res'),

    employee:g('f-emp'),deadline:g('f-dead'),comment:g('f-comm')};

}

 

function updateCalc(){

  const d=formVals();

  const{margin,marginPct,discount}=calcT(d);

  const mc=mColor(marginPct);

  document.getElementById('calc-box').innerHTML=

    `<div class="calc-item"><div class="calc-label">Снижение от НМЦ</div><div class="calc-val">${fmtP(discount)}</div></div>

     <div class="calc-item"><div class="calc-label">Маржинальность</div><div class="calc-val" style="color:${mc}">${fmtP(marginPct)}</div></div>

     ${margin!=null?`<div class="calc-item"><div class="calc-label">Маржа (руб.)</div><div class="calc-val">${fmtR(margin)}</div></div>`:''}`;

}

 

function buildModal(d={},mode='add'){

  const v=(k,def='')=>esc(d[k]??def);

  const opts=(arr,cur)=>arr.map(o=>`<option${(d[o]||cur)===o?' selected':''}>${esc(o)}</option>`).join('');

  // for selects with default val

  const optsD=(arr,cur)=>arr.map(o=>`<option${cur===o?' selected':''}>${esc(o)}</option>`).join('');

  const inpH='class="inp"';

  return`

  <div class="modal-header">

    <span class="modal-title">${mode==='add'?'Добавить тендер':'Редактировать тендер'}</span>

    <button class="modal-close" onclick="closeModal()">✕</button>

  </div>

  <div class="modal-body">

    <div class="field"><label class="field-label">Номер закупки</label>

      <input id="f-num" ${inpH} value="${v('number')}" placeholder="0123456789..."></div>

    <div class="field"><label class="field-label">Тип закупки</label>

      <select id="f-type" ${inpH}>${optsD(TYPES,d.type||TYPES[0])}</select></div>

    <div class="field field-full"><label class="field-label">Наименование закупки</label>

      <input id="f-name" ${inpH} value="${v('name')}" placeholder="Поставка, выполнение работ..."></div>

    <div class="field"><label class="field-label">Заказчик</label>

      <input id="f-cust" ${inpH} value="${v('customer')}" placeholder="Наименование заказчика"></div>

    <div class="field"><label class="field-label">Торговая площадка</label>

      <select id="f-plat" ${inpH}>${optsD(PLATFORMS,d.platform||PLATFORMS[0])}</select></div>

    <div class="field field-full"><label class="field-label">Ссылка на закупку</label>

      <input id="f-url" ${inpH} type="url" value="${v('url')}" placeholder="https://..."></div>

    <div class="field"><label class="field-label">НМЦ (руб.)</label>

      <input id="f-nmc" ${inpH} type="number" min="0" value="${v('nmc')}" oninput="updateCalc()"></div>

    <div class="field"><label class="field-label">Наша цена (руб.)</label>

      <input id="f-price" ${inpH} type="number" min="0" value="${v('price')}" oninput="updateCalc()"></div>

    <div class="field"><label class="field-label">Себестоимость (руб.)</label>

      <input id="f-cost" ${inpH} type="number" min="0" value="${v('cost')}" oninput="updateCalc()"></div>

    <div id="calc-box" class="calc-box"><div style="color:#9ca3af;font-size:12px">Заполните цены для расчёта</div></div>

    <div class="field"><label class="field-label">Статус</label>

      <select id="f-stat" ${inpH}>${optsD(STATUSES,d.status||STATUSES[0])}</select></div>

    <div class="field"><label class="field-label">Решение</label>

      <select id="f-dec" ${inpH}>${optsD(DECISIONS,d.decision||DECISIONS[0])}</select></div>

    <div class="field"><label class="field-label">Итог</label>

      <select id="f-res" ${inpH}>${optsD(RESULTS,d.result||RESULTS[0])}</select></div>

    <div class="field"><label class="field-label">Сотрудник (разместил заявку)</label>

      <input id="f-emp" ${inpH} value="${v('employee',currentUser)}" placeholder="Фамилия Имя"></div>

    <div class="field"><label class="field-label">Срок подачи заявок</label>

      <input id="f-dead" ${inpH} type="date" value="${v('deadline')}"></div>

    <div class="field field-full"><label class="field-label">Комментарий</label>

      <textarea id="f-comm" class="inp" style="height:60px">${v('comment')}</textarea></div>

  </div>

  <div class="modal-footer">

    <button class="btn btn-secondary" onclick="closeModal()">Отмена</button>

    <button class="btn btn-primary" onclick="saveModal('${mode}',${d.id||'null'})">${mode==='add'?'Добавить':'Сохранить'}</button>

  </div>`;

}

 

function openAdd(){

  document.getElementById('modal-box').innerHTML=buildModal({employee:currentUser},'add');

  document.getElementById('modal-overlay').style.display='flex';

  updateCalc();

}

function openEdit(id){

  const t=tenders.find(x=>x.id===id);

  if(!t)return;

  document.getElementById('modal-box').innerHTML=buildModal(t,'edit');

  document.getElementById('modal-overlay').style.display='flex';

  updateCalc();

}

function closeModal(){document.getElementById('modal-overlay').style.display='none';}

 

async function saveModal(mode,id){

  const d=formVals();

  closeModal();

  if(mode==='add')await insert(d);

  else await update(id,d);

}

function closeDelete(){deleteId=null;document.getElementById('del-overlay').style.display='none';}

async function confirmDelete(){if(deleteId){await remove(deleteId);closeDelete();}}

function exportXlsx(){

      'Маржинальность (%)':marginPct!=null?+marginPct.toFixed(2):'',

      'Статус':t.status,'Решение':t.decision,'Итог':t.result,

      'Сотрудник':t.employee,'Срок подачи':t.deadline,'Ссылка':t.url,'Комментарий':t.comment,

    };

  });

  const ws=XLSX.utils.json_to_sheet(rows);

  // Column widths

  ws['!cols']=[20,35,25,18,10,16,16,16,14,14,18,12,12,20,14,35,30].map(w=>({wch:w}));

  const wb=XLSX.utils.book_new();

  XLSX.utils.book_append_sheet(wb,ws,'Тендеры');

  XLSX.writeFile(wb,`ТехноПро_тендеры_${new Date().toISOString().slice(0,10)}.xlsx`);

}
